import tkinter as tk

from tkinter import filedialog

import json

def load_config():

with open('config.json', 'r') as file:

return json.load(file)

def convert_text_to_html(input_text):

lines = input_text.split('\n')

content_html = []

sidebar_links = []

step_counter = 0

article_open = False

main_title = ""

for line in lines:

if line.startswith('#####'):

main_title = line[5:].strip()

elif line.startswith('####'):

link_text = line[4:].strip()

sidebar_links.append(f'{link_text}')

elif line.startswith('###'):

if article_open:

content_html.append('')

article_open = True

content_html.append(f'

')

content_html.append(f'

{line[3:].strip()}

')

step_counter += 1

elif line.startswith('##'):

content_html.append(f'

{line[2:].strip()}

')

elif line.strip():

content_html.append(f'

{line.strip()}

')

if article_open:

content_html.append('

')

sidebar_html = '

\n'

if main_title:

sidebar_html += f'

{main_title}

\n'

sidebar_html += '\n'.join(sidebar_links)

sidebar_html += '\n

'

full_html = f'''

{main_title}

stelarn pages

  • Home

  • About

  • Contact

{sidebar_html}

{''.join(content_html)}

© 2025 stelarn pages. All rights reserved.

'''

return full_html

def submit_action():

input_text = text_box.get("1.0", tk.END)

html_content = convert_text_to_html(input_text)

save_path = filedialog.asksaveasfilename(defaultextension=".html", filetypes=[("HTML files", "*.html")])

if save_path:

with open(save_path, 'w') as file:

file.write(html_content)

status_label.config(text=f"File saved: {save_path}")

config = load_config()

root = tk.Tk()

root.title("Text to HTML Converter")

# Set the window background color from config

root.configure(bg=config['window_bg'])

# Create a frame to hold the text box and status label

frame = tk.Frame(root, bg=config['window_bg'])

frame.pack(fill='both', expand=True, padx=10, pady=10)

# Create a text box with dark mode styling

text_box = tk.Text(frame, wrap='word', bg=config['text_bg'], fg=config['text_fg'], insertbackground='white', font=('Arial', 12))

text_box.pack(fill='both', expand=True, padx=5, pady=5)

# Create a submit button with dark mode styling

submit_button = tk.Button(frame, text="Submit", command=submit_action, bg=config['button_bg'], fg=config['button_fg'], font=('Arial', 12))

submit_button.pack(pady=5)

# Create a status label with dark mode styling

status_label = tk.Label(frame, text="", bg=config['status_bg'], fg=config['status_fg'], font=('Arial', 10))

status_label.pack(pady=5)

root.mainloop()
